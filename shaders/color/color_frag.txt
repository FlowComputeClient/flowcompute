#version 450

// Input variable from the vertex shader
layout(location = 0) in float texCoord;

// The 1D texture sampler mapped to m_colorMapImageInfo
layout(binding = 1) uniform sampler1D colormap;

// Output color
layout(location = 0) out vec4 outColor;

void main() {
    outColor = texture(colormap, clamp(texCoord, 0.0, 1.0));
}
